import { Block, ImageBlock, parseRoot } from "codehike/blocks"

          </div>

import { Block, ImageBlock, parseRoot } from "codehike/blocks"

const Schema = Block.extend({

      talk: Block,

export default function Conf() {

    <main>

      </section>

            </span>
          </div>
        ))}

  )
