function Section(props: {
  steps: Array<{
    title: string
    children: React.ReactNode
    screenshot?: Image
    code?: Code
  }>
}) {
  const [index, setIndex] = useState(0)
  const step = props.steps[index]
  return (
    <section>
      <Observer steps={steps} onChange={setIndex} />
      <Sticky step={step} />
    </section>
  )
}
```
